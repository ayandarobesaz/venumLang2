expresion   -> literal
             | unary
             | binary
             | grouping;

literal     -> NUMBER | STRING | "true" | "false" | "nil";
grouping    -> "(" expresion ")";
unary       -> ( "-" | "!" ) expresion;
binary      -> expresion operator expresion;
operator    -> "==" | "!=" | "<" | "<=" | ">" | ">=" | "+" | "-" | "*" | "/";
